
//////////////////////////////////////////////////////////////
// TESTBENCH
//////////////////////////////////////////////////////////////
`timescale 1ns / 1ps

module Traffic_Light_Controller_TB;

reg clk;
reg rst;

reg [1:0] density_M1;
reg [1:0] density_M2;
reg [1:0] density_S;

reg emergency;
reg pedestrian_btn;
reg night_mode;

wire [2:0] light_M1;
wire [2:0] light_S;
wire [2:0] light_MT;
wire [2:0] light_M2;

wire [6:0] seg;
wire [3:0] an;
wire tx;

//============================================================
// DUT
//============================================================

top_module dut(

    .clk(clk),
    .rst(rst),
    .tx(tx),

    .density_M1(density_M1),
    .density_M2(density_M2),
    .density_S(density_S),

    .emergency(emergency),
    .pedestrian_btn(pedestrian_btn),
    .night_mode(night_mode),

    .light_M1(light_M1),
    .light_S(light_S),
    .light_MT(light_MT),
    .light_M2(light_M2),

    .seg(seg),
    .an(an)

);

//============================================================
// CLOCK
//============================================================

always #5 clk = ~clk;

//============================================================
// TEST SEQUENCE
//============================================================

initial
begin

    //////////////////////////////////////////////////////////
    // INITIAL VALUES
    //////////////////////////////////////////////////////////

    clk = 0;
    rst = 1;

    density_M1 = 2'b11;
    density_M2 = 2'b01;
    density_S  = 2'b00;

    emergency = 0;
    pedestrian_btn = 0;
    night_mode = 0;

    //////////////////////////////////////////////////////////
    // RESET
    //////////////////////////////////////////////////////////

    #20;
    rst = 0;

    //////////////////////////////////////////////////////////
    // NORMAL MODE
    //////////////////////////////////////////////////////////

    #100;

    //////////////////////////////////////////////////////////
    // EMERGENCY MODE
    //////////////////////////////////////////////////////////

    emergency = 1;

    #50;

    emergency = 0;

    //////////////////////////////////////////////////////////
    // PEDESTRIAN MODE
    //////////////////////////////////////////////////////////

    #100;

    pedestrian_btn = 1;

    #20;

    pedestrian_btn = 0;

    //////////////////////////////////////////////////////////
    // NIGHT MODE
    //////////////////////////////////////////////////////////

    #100;

    night_mode = 1;

    #100;

    night_mode = 0;

    //////////////////////////////////////////////////////////
    // RUN MORE
    //////////////////////////////////////////////////////////

    #500;

    $finish;

end

endmodule
